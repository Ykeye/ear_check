function showResults() { document.getElementById("gameScreen").classList.add("hidden"); document.getElementById("resultScreen").classList.remove("hidden"); let list = document.getElementById("resultsList"); list.innerHTML = ""; let score = 0; rounds.forEach((r, i) => { if (r.correctFlag) score++; let diffHz = Math.abs(r.f1 - r.f2).toFixed(2); let card = document.createElement("div"); card.style.margin = "12px auto"; card.style.padding = "14px"; card.style.maxWidth = "420px"; card.style.borderRadius = "12px"; card.style.background = r.correctFlag ? "#12351f" : "#3a1a1a"; card.style.border = r.correctFlag ? "1px solid #2ecc71" : "1px solid #e74c3c"; card.style.textAlign = "left"; card.innerHTML = `
Round ${i + 1}
Correct: ${r.correct.toUpperCase()}
Your answer: ${r.user ? r.user.toUpperCase() : "—"}
${r.correctFlag ? "✅ Correct" : "❌ Wrong"}
${r.cents} cents | A: ${r.f1.toFixed(2)} Hz | B: ${r.f2.toFixed(2)} Hz | Δ ${diffHz} Hz
`; list.appendChild(card); }); document.getElementById("score").innerText = `Score: ${score} / 5`; }
